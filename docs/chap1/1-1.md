  created() {},
  methods: {
    runBase(name) {
      this.clear(name);
      let str = this.codes[name];
      let display = (str) => {
        this.msg[name] = this.msg[name] + str;
      };
      run(str,display,false,this.env)
    },
    runSub(name) {
      this.clear(name);
      let str = this.codes[name];
      let display = (str) => {
        this.msg[name] = this.msg[name] + str;
      };
      run(str,display,true,this.env)
    },
    clear(name) {
      this.msg[name] = "";
    },
  },
};
